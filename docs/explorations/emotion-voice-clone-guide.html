cnn_lstm_model.py
import tensorflow as tf
from tensorflow.keras import layers, Model

def create_cnn_lstm_model(vocab_size, embedding_dim, max_length):
    # Input layer
    inputs = layers.Input(shape=(max_length,))

    # Hindi word embeddings (pre-trained or trainable)
    embedding = layers.Embedding(vocab_size, embedding_dim)(inputs)

    # CNN for spatial feature extraction
    conv1 = layers.Conv1D(256, 5, activation='relu')(embedding)
    conv1 = layers.MaxPooling1D(5)(conv1)
    conv2 = layers.Conv1D(128, 3, activation='relu')(conv1)
    conv2 = layers.MaxPooling1D(5)(conv2)

    # LSTM for temporal sequence modeling
    lstm = layers.Bidirectional(layers.LSTM(128, return_sequences=True))(conv2)
    lstm = layers.Bidirectional(layers.LSTM(64))(lstm)

    # Dense layers
    dense = layers.Dense(64, activation='relu')(lstm)
    dropout = layers.Dropout(0.5)(dense)

    # Output layer (7 emotions)
    outputs = layers.Dense(7, activation='softmax')(dropout)
    return Model(inputs=inputs, outputs=outputs)

import re
from indicnlp.tokenize import indic_tokenize

def preprocess_hindi_text(text):
    # Remove special chars, preserve Hindi diacritics
    text = re.sub(r'[^\u0900-\u097F\s\w]', '', text)

    # Tokenize Hindi text
    tokens = indic_tokenize.trivial_tokenize(text)

    # Remove Hindi-specific stopwords
    hindi_stopwords = load_hindi_stopwords()
    tokens = [t for t in tokens if t not in hindi_stopwords]
    return ' '.join(tokens)

def create_hinglish_embeddings():
    # Mix Hindi and English embeddings
    hindi_emb = load_hindi_word2vec()

# Evaluation metrics for Hindi emotion recognition
from sklearn.metrics import classification_report

def evaluate_model(model, test_data, test_labels):
    predictions = model.predict(test_data)
    predicted_classes = tf.argmax(predictions, axis=1)
    true_classes = tf.argmax(test_labels, axis=1)

    # Hindi emotion labels
    emotion_labels = ['खुशी', 'दुःख', 'क्रोध',
                     'डर', 'आश्चर्य', 'घृणा', 'तटस्थ']

    report = classification_report(true_classes, predicted_classes,
                                 target_names=emotion_labels)
    return report

def detect_emotion_stream(text_stream):
    for text in text_stream:
        if is_hindi(text) or is_hinglish(text):
            emotion = predict_single(text)
            # Add cultural context for Indian emotions

                if contains_festival_words(text):
                    emotion = 'festive_happiness'
                elif contains_family_words(text):
                    emotion = 'family_happiness'
            yield {"text": text, "emotion": emotion}

# Hindi/English code-switching detection
def detect_code_switching(text):
    hindi_words = extract_hindi_words(text)
    english_words = extract_english_words(text)
    ratio = len(english_words) / (len(hindi_words) + 1)
    if ratio < 0.1: return "pure_hindi"
    elif ratio < 0.4: return "hinglish_light"
    else: return "hinglish_heavy"
 Summary

Complete Architecture Summary

Input: Hindi/Hinglish text → Preprocessing: Indic NLP tokenization, stopword removal → Embedding: Hindi Word2Vec + FastText mixed embeddings → CNN Layers: 3 Conv1D layers with ReLU activation → Pooling: Global max pooling → LSTM Layers: Bi-directional LSTM (2 layers) → Fusion: Concatenate CNN + LSTM outputs → Dense Layers: 2 fully connected layers with dropout → Output: 7 emotion probabilities with softmax.

Key Innovations:

  • Mixed embeddings for Hindi-English code-switching
  • CNN captures n-gram patterns in Devanagari script
  • LSTM models sentence structure and context
  • Attention mechanism for important emotion words
  • Cultural emotion mapping for Indian context
